s = "无可挑剔"

encoder = base64.b64encode(s.encode("utf-8"))
print(encoder)

str_encoder = encoder.decode('utf-8')
print(str_encoder)

# 解码过程
decoder = base64.b64decode(str_encoder)
print(decoder.decode('utf-8'))
